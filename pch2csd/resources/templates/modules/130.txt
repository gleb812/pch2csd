  	kCount += 1
  	zkw int(kCount*.5), 1 ; CHANGE 
  	zkw int(kCount*.25), 1 ; CHANGE 
  	zkw int(kCount*.125), 1 ; CHANGE 
  	zkw int(kCount*.0625), 1 ; CHANGE

  	zkw int(kCount*.015625), 1 ; CHANGE 
  	zkw int(kCount*.0078125), 1 ; CHANGE 
  	zkw int(kCount*.00390625), 1 ; CHANGE
  endif

  	kCount = 0 
  endif
