;@ ins k
;@ outs k k
opcode swOnOffM, 0, kkkk
  kSw, kin, kout, kctrl xin

  if kin != 0 goto Next
      kIn = 1
      goto Over
  Next:
  kIn zkr kin
  Over:

  zkw kIn*kSw, kout
  zkw kSw/16,  kctrl
endop

  if kin != 0 goto Next
      kIn = 1
  Next:

  zkw kSw/16,  kctrl
endop
